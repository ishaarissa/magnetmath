document.querySelectorAll('.number, .token').forEach(el => {
  el.id = 'drag-' + Math.random().toString(36).substr(2, 9);
  el.addEventListener('dragstart', ev => {
    ev.dataTransfer.setData("text", ev.target.id);
  });
});

function allowDrop(ev) { ev.preventDefault(); }

function drop(ev) {
  ev.preventDefault();
  const id = ev.dataTransfer.getData("text");
  const draggedEl = document.getElementById(id);
  ev.target.appendChild(draggedEl);
}

document.getElementById('checkBtn').addEventListener('click', () => {
  const qInput = document.getElementById('question').value.trim();
  const [num1, num2] = qInput.split('-').map(n => parseInt(n));
  if (isNaN(num1) || isNaN(num2)) {
    document.getElementById('result').textContent = "â— Please enter a valid subtraction question.";
    document.getElementById('result').style.color = "orange";
    return;
  }
  const correctAnswer = num1 - num2;
  const h = document.getElementById('hundreds').innerText.match(/\d+/g);
  const t = document.getElementById('tens').innerText.match(/\d+/g);
  const o = document.getElementById('ones').innerText.match(/\d+/g);
  const studentNum = `${h ? h.join('') : ''}${t ? t.join('') : ''}${o ? o.join('') : ''}`;
  const studentAnswer = parseInt(studentNum);
  if (studentAnswer === correctAnswer) {
    document.getElementById('result').textContent = "âœ… Correct!";
    document.getElementById('result').style.color = "green";
  } else {
    document.getElementById('result').textContent = `âŒ Incorrect. Correct answer: ${correctAnswer}`;
    document.getElementById('result').style.color = "red";
  }
});

document.getElementById('resetBtn').addEventListener('click', () => {
  ['hundreds','tens','ones'].forEach(id => {
    const col = document.getElementById(id);
    while (col.firstChild) { col.removeChild(col.firstChild); }
    const h = document.createElement('h2');
    h.textContent = id.charAt(0).toUpperCase() + id.slice(1);
    col.appendChild(h);
  });
  document.getElementById('result').textContent = '';
});
